#include<iostream>
#include<cmath>  
using namespace std; 
//16160200518 ÎéÕêÀÚ 
class point 
{  
 private:
    float x,y;
 public:
 point(float a,float b) { x=a; y=b; }
   float Distance(point &p)
	{ float dx=p.x-x;
	   float dy=p.y-y;
	   return (float)sqrt(dx*dx+dy*dy);
	}

};

int main() 
{
    point p1(2,3),p2(32,43);
    cout<<p1.Distance(p2)<<endl;
    return 0;
}
